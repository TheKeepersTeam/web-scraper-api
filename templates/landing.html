# Get your API key
curl -X POST https://api.cyberraccoonteam.com/api/keys

# Scrape a website
curl -X POST https://api.cyberraccoonteam.com/api/scrape \
  -H "Content-Type: application/json" \
  -H "X-API-Key: YOUR_KEY" \
  -d '{"url": "https://example.com"}'

# Check usage
curl https://api.cyberraccoonteam.com/api/usage \
  -H "X-API-Key: YOUR_KEY"
